country_event = {
	id = RoF.005
	title = RoF.005.t
	desc = RoF.005.d
	picture = GFX_report_event_scottish_mercenaries
	hidden = yes
	trigger = {
		OR = {
			has_country_flag = colony_flag
			has_country_flag = crown_union_flag
			has_country_flag = federal_state
		}
		is_subject = no
	}
	mean_time_to_happen = {
		days = 30
	}
	
	immediate = {
		if = {
			limit = { has_country_flag = colony_flag }
			set_country_flag = former_colony
			clr_country_flag = colony_flag
		}
		clr_country_flag = crown_union_flag
		clr_country_flag = federal_state
	}
	
	option = {
		name = RoF.005.a
	}
	
}

country_event = {
	id = RoF.010
	title = RoF.010.t
	desc = RoF.010.d
	picture = GFX_report_event_scottish_mercenaries
	
	is_triggered_only = yes
	trigger = {
		any_neighbor_country = {
			has_variable = CoC_strength_@THIS
		}
	}
	
	immediate = {
		hidden_effect = {
			set_country_flag = CoC_present_@ROOT
			set_variable = { var = CoC_strength_@ROOT value = 0 }
			set_variable = { var = CoC_anger_@ROOT value = 0 }
			if = {
				limit = {
					has_government = absolutist
				}
				add_to_variable = { var = CoC_anger_@ROOT value = 5 }
			}
			else_if = {
				limit = {
					has_government = radical
				}
				add_to_variable = { var = CoC_anger_@ROOT value = -10 }				
			}
		}
	}
	
	option = {
		name = RoF.010.a
	}
}
