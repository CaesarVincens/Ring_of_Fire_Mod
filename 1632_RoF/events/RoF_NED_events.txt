# capture of Gaston
country_event = {
	id = RoF_NED.152
	title = RoF_NED.152.t
	desc = RoF_NED.152.d

	picture = GFX_report_event_ups_flag
	
	is_triggered_only = yes
	
	option = {
		name = RoF_NED.152.a
		if = {
			limit = {
				FRA = {
					owns_state = 240
				}
			}
			FRA = { release = LOR }
		}
		if = {
			limit = {
				FRA = {
					owns_state = 241
				}
			}
			LOR = { transfer_state = 241 }
		}
		set_autonomy = {
			target = LOR
			autonomy_state = autonomy_rof_protectorate
		}
	}
	option = {
		trigger = { has_country_flag = BRG_assisting_with_gaston }
		name = RoF_NED.152.b
		if = {
			limit = {
				240 = {
					OR = {
						is_owned_by = FRA
						is_owned_by = LOR
					}
				}
			}
			BRG = { transfer_state = 240 }
		}
		if = {
			limit = {
				241 = {
					OR = {
						is_owned_by = FRA
						is_owned_by = LOR
					}
				}
			}
			SPN = { transfer_state = 241 }
		}
	}
}
