#Don Fernando takes control of SPN
country_event = {
	id = RoF_NED.102
	title = RoF_NED.102.t
	picture = GFX_report_event_spr_flag
	desc = RoF_NED.102.d
	fire_only_once = yes

	trigger = {
		tag = SPN
		has_country_flag = SPN_don_fernando_arrives
	}
	
	mean_time_to_happen = {
		days = 30
	}
	
	immediate = {
		hidden_effect = {
			create_country_leader = {
				name = "Fernando von Habsburg"
				desc = "POLITICS_DON_FERNANDO_DESC"
				picture = "GFX_leader_SPN_Don_Fernando"
				ideology = absolute_monarchy
				traits = {
				}
			}
			create_country_leader = {
				name = "Fernando von Habsburg"
				desc = "POLITICS_DON_FERNANDO_DESC"
				picture = "GFX_leader_SPN_Don_Fernando"
				ideology = constitutional_monarchy
				traits = {
				}
			}
		}
	}
	
	option = {
		name = RoF_NED.102.a
		if = {
			limit = { NOT = { has_government = constitutionalist } }
			add_popularity = { ideology = constitutionalist popularity = 0.4 }
			set_politics = { ruling_party = constitutionalist }
		}
		effect_tooltip = {
			create_country_leader = {
				name = "Fernando von Habsburg"
				desc = "POLITICS_DON_FERNANDO_DESC"
				picture = "GFX_leader_SPN_Don_Fernando"
				ideology = constitutional_monarchy
				traits = {
				}
			}		
		}
		ai_chance = {
			factor = 10
			modifier = {
				is_historical_focus_on = yes
				factor = 10
			}
		}
	}
	option = {
		name = RoF_NED.102.b
		if = {
			limit = { NOT = { has_government = absolutist } }
			add_popularity = { ideology = absolutist popularity = 0.4 }
			set_politics = { ruling_party = absolutist }
		}
		effect_tooltip = {
			create_country_leader = {
				name = "Fernando von Habsburg"
				desc = "POLITICS_DON_FERNANDO_DESC"
				picture = "GFX_leader_SPN_Don_Fernando"
				ideology = absolute_monarchy
				traits = {
				}
			}		
		}
		ai_chance = {
			factor = 5
		}
	}
}

# SPN gobbles up border states
country_event = {
	id = RoF_NED.115
	title = RoF_NED.115.t
	desc = RoF_NED.115.d
	picture = GFX_report_event_SPR_flag
	
	is_triggered_only = yes
	
	option = { # Accept
		name =  RoF_NED.115.a
		set_country_flag = @ROOT_accept_demand
		SPN = {
			country_event = { id = RoF_NED.116 }
		}
		custom_effect_tooltip = GAME_OVER_TT
		ai_chance = {
			base = 2
			modifier = {
				factor = 10
				NOT = {
					strength_ratio = {
						tag = SPN
						ratio = 0.75
					}
				}
			}
		}
	}
	option = { # Refuse
		name =  RoF_NED.115.b
		set_country_flag = @ROOT_refuse_demand
		every_owned_state = {
			add_claim_by = SPN
		}
		SPN = {
			country_event = { id = RoF_NED.116 }
		}
		ai_chance = {
			base = 1
		}
	}
}

		trigger = {
			FROM = { has_country_flag = @FROM_refuse_demand }
		}
	}

		text = RoF_NED.116.db
		trigger = {
			FROM = { has_country_flag = @FROM_accept_demand }
		}
	}
	picture = GFX_report_event_SPR_flag
	
	is_triggered_only = yes
	
	option = {
		name = RoF_NED.116.a
		trigger = {
			FROM = { has_country_flag = @FROM_refuse_demand }

			target = FROM
			type = annex_everything
		}
	}
	option = {
		name = RoF_NED.116.b
		trigger = {
			FROM = { has_country_flag = @FROM_accept_demand }
		}
		annex_country = { target = FROM }
	}
}
