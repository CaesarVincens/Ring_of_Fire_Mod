RoF_enough_spears = {
	name = "enough spears"
	desc = "you have enough"
	enable = {
		has_equipment = {
			pike_equipment > 50000
		}
	}
	abort_when_not_enabled = yes
	ai_strategy = {
		type = equipment_variant_production_factor

		value = -2000
	}
}
