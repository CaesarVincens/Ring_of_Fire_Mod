# Allows you to create your own dynamic keys
# to be called in localization.
# defined_text -> this is it, we're defining the text
# text -> a discrete entry that can be picked to display in loc.
# trigger -> determines if a text entry will be picked or not.
# (The triggers need to be valid for the scope the key is called in
# (eg Root or From.From).)
# localization_key -> points to the localization key
# that'll be used if trigger passes

#pseudocode for now until ideas are done

defined_text = { # Country
	name = GetCoCStrength

	text = {
		trigger = {
			check_variable = { var = CoC_strength_@THIS value = 20 compare = greater_than_or_equals }
		}
		localization_key = ROF_COC_DOMINANT
	}

	text = {
		trigger = {
			check_variable = { var = CoC_strength_@THIS value = 15 compare = greater_than_or_equals }
			check_variable = { var = CoC_strength_@THIS value = 20 compare = less_than }
		}
		localization_key = ROF_COC_STRONG
	}

	text = {
		trigger = {
			check_variable = { var = CoC_strength_@THIS value = 10 compare = greater_than_or_equals }
			check_variable = { var = CoC_strength_@THIS value = 15 compare = less_than }
		}
		localization_key = ROF_COC_INFLUENTIAL
	}

	text = {
		trigger = {
			check_variable = { var = CoC_strength_@THIS value = 5 compare = greater_than_or_equals }
			check_variable = { var = CoC_strength_@THIS value = 10 compare = less_than }
		}
		localization_key = ROF_COC_WEAK
	}

	text = {
		trigger = {
			check_variable = { var = CoC_strength_@THIS value = 0 compare = greater_than_or_equals }
			check_variable = { var = CoC_strength_@THIS value = 5 compare = less_than }
		}
		localization_key = ROF_COC_INSIGNIFICANT
	}

	text = {
		trigger = {
			check_variable = { var = CoC_strength_@THIS value = 0 compare = less_than }
		}
		localization_key = ROF_COC_NON_EXISTANT
	}

}
