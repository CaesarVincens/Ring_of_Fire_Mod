		has_capitulated = no
	}

is_friend_or_potential_friend_of_ROOT = {
	custom_trigger_tooltip = {
		tooltip = is_friend_or_potential_friend_of_ROOT_tt
		OR = {
			is_subject_of = ROOT
			is_subject = no
			is_in_faction_with = ROOT
		}
		OR = {
			is_in_faction_with = ROOT
			is_in_faction = no
		}
		NOT = { has_war_with = ROOT }
		has_capitulated = no
		exists = yes
	}
}

		}
	}
}
