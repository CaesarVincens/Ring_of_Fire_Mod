TUR_save_admiral_pp_for_later = {
	enable = {
		original_tag = TUR
		has_war = no
		threat < 0.3
		has_political_power < 200		
	}
	abort = {
		OR = {
			has_war = yes	
			threat > 0.3
			has_political_power > 200
		}
	}

	ai_strategy = { # dont burn PP on admirals (can still get free ones)
		type = pp_spend_priority
		id = admiral
		value = -100
	}
	ai_strategy = {
		type = pp_spend_priority
		id = general
		value = -100
	}
}
