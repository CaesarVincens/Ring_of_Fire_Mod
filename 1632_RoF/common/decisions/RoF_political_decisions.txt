			hidden_trigger = {

		days_remove = 30

		modifier = {
			absolutist_drift = 0.02
			stability_weekly = -0.03
 			political_power_gain = -0.2
		}
		
		ai_will_do = {
			factor = 1
		}

		modifier = {
			command_power_gain = -0.2
		}

		remove_effect = {
			if = {
				limit = {
					check_variable = { var = army_support_var value = 0.1 compare = less_than }
				}
				set_variable = { var = army_support_var value = 0.1 }
			}
			if = {
				limit = {
					check_variable = { var = army_support_var value = 0.1 compare = greater_than_or_equals }
				}
				add_to_variable = { var = army_support_var value = 0.1 }
			}
			set_country_flag = military_support_for_absolutist_civil_war
			custom_effect_tooltip = RoF_army_support_TT
		}
		
		complete_effect = {
		}
		
		ai_will_do = {
			factor = 0
		}
	}

	ignite_the_absolutist_civil_war = {

		icon = generic_ignite_civil_war

		available = {
			has_stability < 0.5
		}

		visible = {
			has_country_flag = preparation_for_absolutist_civil_war
			has_idea_with_trait = absolutist_minister
			#check_has_focus_tree_to_switch_to_absolutist = yes
			NOT = { has_government = absolutist }
		}
	
		fire_only_once = yes
	
		cost = 50
		
		complete_effect = {
			if = {
				limit = {
					has_government = constitutionalist
				}
				set_variable = {
					var = civil_war_size_var
					value = party_popularity@constitutionalist
				}
			}
			if = {
				limit = {
					has_government = liberal
				}
				set_variable = {
					var = civil_war_size_var
					value = party_popularity@liberal
				}
			}
			if = {
				limit = {
					has_government = radical
				}
				set_variable = {
					var = civil_war_size_var
					value = party_popularity@radical
				}
			}
			if = {
				limit = {
					has_government = neutrality
				}
				set_variable = {
					var = civil_war_size_var
					value = party_popularity@neutrality
				}
			}
			subtract_from_variable = {
				var = civil_war_size_var
				value = army_support_var
			}
			if = {
				limit = {
					check_variable = {
						var = civil_war_size_var
						value = 0.3
						compare = less_than
					}
				}
				set_variable = {
					var = civil_war_size_var
					value = 0.3
				}
			}
			start_civil_war = {
				ruling_party = absolutist
				ideology = ROOT
				size = civil_war_size_var
				keep_unit_leaders_trigger = {
					has_trait = hidden_sympathies
				}
			}
			clr_country_flag = preparation_for_absolutist_civil_war
			clr_country_flag = military_support_for_absolutist_civil_war
			clr_country_flag = civil_support_for_absolutist_civil_war
			set_country_flag = ideology_civil_war
		}
		
		ai_will_do = {
			factor = 0
		}
	}

			has_government = absolutist

			add_stability = -0.010
			if = {

				}

					ideology = constitutionalist

				}
			}
			if = {
				limit = {
					has_government = liberal
				}
				add_popularity = {
					ideology = liberal
					popularity = -0.1
				}

			if = {
				limit = {
					has_government = radical
				}
				add_popularity = {
					ideology = radical
					popularity = -0.1
				}
			}
			if = {
				limit = {
					has_government = neutrality
				}
				add_popularity = {
					ideology = neutrality
					popularity = -0.1
				}
			}
		}
		
		ai_will_do = {
			factor = 1
		}
	}

		}
		
		fire_only_once = yes

		ai_will_do = {
			factor = 210

				absolutist > 0.6
				factor = 10
			}
		}
